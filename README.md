с ключем проделываем тоже самое, только до 1000 символов

разбиваем текст на кучу кусочков длины ключа

в начало привисаывем ключ
m - кол-во блоков без ключа
n - кол-во проходов

key -> arr[0] => sh[0]
sh[0] -> arr[1] => sh[1]
sh[1] -> arr[2] => sh[2]
		...
sh[m - 1] -> arr[0] => sh[m]
		...
sh[mn - 2] -> arr[m - 1] => sh[mn - 1]
